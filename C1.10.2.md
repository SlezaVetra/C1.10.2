class Rectangle:
    def init(self, width, height):
        self.width = width
        self.height = height

    def get_width(self):
        return self.width

    def get_height(self):
        return self.height

    def get_area(self):
        return self.width*self.height


print(f"{rectangle.get_width()},  {rectangle.get_height()}, {rectangle.get_area()}")
